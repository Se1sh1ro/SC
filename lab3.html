import os from dotenv import load_dotenv import anthropic import time load_dotenv() client = anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY")) # Сравнение моделей models = ["claude-3-haiku-20240307", "claude-3-sonnet-20240229"] prompt = "Объясни, что такое API, простыми словами" for model in models: start = time.time() response = client.messages.create( model=model, max_tokens=200, temperature=0.3, messages=[{"role": "user", "content": prompt}] ) elapsed = time.time() - start print(f"\n{model} ({elapsed:.2f}с):\n{response.content[0].text}") # Тест max_tokens print("\n--- Тест max_tokens ---") for tokens in [10, 50, 100, 500]: response = client.messages.create( model="claude-3-haiku-20240307", max_tokens=tokens, temperature=0, messages=[{"role": "user", "content": "Напиши подробный рассказ о космосе"}] ) print(f"max_tokens={tokens}: {len(response.content[0].text)} символов") # Тест истории сообщений print("\n--- Тест истории сообщений ---") messages = [] messages.append({"role": "user", "content": "Привет! Меня зовут Анна"}) response = client.messages.create( model="claude-3-haiku-20240307", max_tokens=100, temperature=0, messages=messages ) messages.append({"role": "assistant", "content": response.content[0].text}) print(f"Бот: {response.content[0].text}") messages.append({"role": "user", "content": "Как меня зовут?"}) response = client.messages.create( model="claude-3-haiku-20240307", max_tokens=100, temperature=0, messages=messages ) print(f"Бот: {response.content[0].text}")
