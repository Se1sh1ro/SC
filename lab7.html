import os from dotenv import load_dotenv import anthropic load_dotenv() client = anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY")) # Задача без CoT print("=== Без CoT ===") response = client.messages.create( model="claude-3-haiku-20240307", max_tokens=500, temperature=0.2, messages=[{"role": "user", "content": "У Маши было 15 яблок. Она отдала треть Пете, а затем купила ещё 5. Сколько яблок у Маши?"}] ) print(response.content[0].text) # Задача с CoT print("\n=== С CoT ===") response = client.messages.create( model="claude-3-haiku-20240307", max_tokens=500, temperature=0.2, messages=[{"role": "user", "content": "Реши задачу по шагам.\nУ Маши было 15 яблок. Она отдала треть Пете, а затем купила ещё 5.
